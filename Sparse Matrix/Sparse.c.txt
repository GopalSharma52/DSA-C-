#include <stdio.h>

int isSparse(int rows,int cols,int matrix[rows][cols]){
    int zerocount = 0;
    for(int i = 0; i<rows; i++){
        for(int j = 0; j<cols; j++){
            if(matrix[i][j]==0){
                zerocount++;
            }
        }
    }
    return zerocount>(rows*cols)/2;

}


int main(){
    int matrix[3][3] = {{0,2,3},{4,5,0},{1,0,9}};
    
    if(isSparse(3,3,matrix)){
        printf("Matrix is sparse: ");
    }else{
        printf("Matrix is not sparse");
    }
    return 0;

}
